Maven Repository

pom.xml(repository) setting:


	<repositories>
		<repository>
			<id>OpenIG Standalone Repository</id>
			<name>OpenIG Standalone Extension Repository</name>
			<url>http://alunko.github.com/standalone-openig/repository/</url>
			<layout>default</layout>
		</repository>
	</repositories>
	<dependencies>
		<dependency>
			<groupId>com.github.standalone-openig</groupId>
			<artifactId>standalone-openig</artifactId>
			<version>1.4</version>
			<scope>compile</scope>
		</dependency>
	</dependencies>
